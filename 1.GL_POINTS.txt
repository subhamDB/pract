 #include<GL/glut.h>
 void display()
 {
 glClear(GL_COLOR_BUFFER_BIT);//glClear actually clears the buffer,
 glColor3f(1.0, 0.0, 0.0);//used to give each vertex its own color
 glBegin(GL_POINTS);
 glVertex2f(10.0, 10.0);// specifies the x and y coordinates of the vertex,
 glVertex2f(150.0, 80.0);
 glVertex2f(100.0, 20.0);
 glVertex2f(200.0, 100.0);
 glEnd();
 glFlush();//all issued commands to be executed as quickly as they are accepted 
 }
 void myinit()
 {
 glClearColor(1.0, 1.0, 1.0, 1.0);//specifies the red, green, blue, and alpha values
 used by glClear to clear the color buffers.
 glColor3f(1.0, 0.0, 0.0);//Sets the current color
 glPointSize(10.0);
 gluOrtho2D(0.0, 499.0, 0.0, 499.0);
 }

 int main(int argc, char** argv)
 {
 glutInit(&argc, argv);
 glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
 glutInitWindowSize(500, 500);
 glutInitWindowPosition(0, 0);
 glutCreateWindow("Points");
 glutDisplayFunc(display);
 myinit();
 glutMainLoop();
 }
