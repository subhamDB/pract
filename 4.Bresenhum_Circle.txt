#include <iostream>
#include <cmath>
#include <GL/glut.h>

using namespace std;

int xc = 320, yc = 240; // Circle center
int radius;            

// Plot symmetric points for each calculated point
void plot_point(int x, int y) {
    glBegin(GL_POINTS);
        glVertex2i(xc + x, yc + y);
        glVertex2i(xc + x, yc - y);
        glVertex2i(xc - x, yc + y);
        glVertex2i(xc - x, yc - y);
        glVertex2i(xc + y, yc + x);
        glVertex2i(xc + y, yc - x);
        glVertex2i(xc - y, yc + x);
        glVertex2i(xc - y, yc - x);
    glEnd();
}

// Bresenham Circle Drawing
void bresenham_circle(int r) {
    int x = 0, y = r;
    int d = 3 - 2 * r; // Initial decision parameter

    while (x <= y) {
        plot_point(x, y);
        if (d < 0) {
            d = d + 4 * x + 6;
        } else {
            d = d + 4 * (x - y) + 10;
            y--;
        }
        x++;
    }

    glFlush();
}

// Display function
void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    bresenham_circle(radius); 
}

// OpenGL Initialization
void Init() {
    glClearColor(1.0, 1.0, 1.0, 0.0);  // White background
    glColor3f(0.0, 0.0, 0.0);          // Black drawing color
    gluOrtho2D(0, 640, 0, 480);        // Set coordinate system
}

// Main function
int main(int argc, char** argv) {
    cout << "Enter the radius of the circle: ";  
    cin >> radius;                               

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(640, 480);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Bresenham Circle Drawing");
    Init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
