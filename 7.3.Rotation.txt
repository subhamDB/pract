#include <iostream>
#include <vector>
#include <GL/glut.h>
#include <cmath>

using namespace std;

int pntX1, pntY1, edges;
vector<int> pntX;
vector<int> pntY;
double angleRad;

void drawPolygon()
{
    glBegin(GL_POLYGON);
    glColor3f(1.0, 0.0, 0.0); // Red color for the original polygon
    for (int i = 0; i < edges; i++)
    {
        glVertex2i(pntX[i], pntY[i]);
    }
    glEnd();
}

void drawPolygonRotation(double angleRad)
{
    glBegin(GL_POLYGON);
    glColor3f(0.0, 0.0, 1.0); // Blue color for rotated polygon
    for (int i = 0; i < edges; i++)
    {
        int xNew = round(pntX[i] * cos(angleRad) - pntY[i] * sin(angleRad));
        int yNew = round(pntX[i] * sin(angleRad) + pntY[i] * cos(angleRad));
        glVertex2i(xNew, yNew);
    }
    glEnd();
}

void myInit(void)
{
    glClearColor(1.0, 1.0, 1.0, 0.0); // Set background to white
    glColor3f(0.0f, 0.0f, 0.0f); // Set drawing color to black
    glPointSize(4.0);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(0.0, 640.0, 0.0, 480.0); // Set the 2D orthogonal projection
}

void myDisplay(void)
{
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(0.0, 0.0, 0.0);
    
    drawPolygon(); // Draw original polygon
    drawPolygonRotation(angleRad); // Draw rotated polygon
    
    glFlush();
}

int main(int argc, char** argv)
{
    int angle;
    cout << "Enter number of edges: "; 
    cin >> edges;

    for (int i = 0; i < edges; i++)
    {
        cout << "Enter coordinates for vertex " << i + 1 << " : "; 
        cin >> pntX1 >> pntY1;
        pntX.push_back(pntX1);
        pntY.push_back(pntY1);
    }

    cout << "Enter the angle of rotation (in degrees): ";
    cin >> angle;
    angleRad = angle * 3.1416 / 180; // Convert angle to radians

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(640, 480);
    glutInitWindowPosition(100, 150);
    glutCreateWindow("Rotation Transformation");
    glutDisplayFunc(myDisplay);
    myInit();
    glutMainLoop();

    return 0;
}
